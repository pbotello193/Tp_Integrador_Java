create database integrador_cac;
use integrador_cac;
CREATE TABLE oradores (
  id_orador INT AUTO_INCREMENT PRIMARY KEY,
  nombre VARCHAR(255),
  apellido VARCHAR(255),
  tema VARCHAR(255),
  fecha_alta date
);
select * from oradores;
